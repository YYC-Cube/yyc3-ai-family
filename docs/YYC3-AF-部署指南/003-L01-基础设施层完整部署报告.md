const WebSocket = require('ws');

const server = http.createServer((req, res) => {
  res.writeHead(200, { 'Content-Type': 'text/plain' });
  res.end('YYC3 Heartbeat Relay');
});

const wss = new WebSocket.Server({ server, path: '/ws/heartbeat' });

wss.on('connection', (ws) => {
  console.log(`[${new Date().toISOString()}] Client connected`);
  
  ws.on('message', (message) => {
    try {
      const data = JSON.parse(message);
      console.log(`[${new Date().toISOString()}] Received:`, data.type);
      
      // å¿ƒè·³å“åº”
      if (data.type === 'heartbeat') {
        ws.send(JSON.stringify({
          type: 'heartbeat-ack',
          timestamp: Date.now(),
          serverTime: new Date().toISOString()
        }));
      }
      
      // å¹¿æ’­æ¶ˆæ¯
      if (data.type === 'broadcast') {
        wss.clients.forEach(client => {
          if (client !== ws && client.readyState === WebSocket.OPEN) {
            client.send(JSON.stringify({
              type: 'broadcast',
              from: data.from,
              payload: data.payload,
              timestamp: Date.now()
            }));
          }
        });
      }
    } catch (e) {
      console.error('Parse error:', e.message);
    }
  });
  
  ws.on('close', () => {
    console.log(`[${new Date().toISOString()}] Client disconnected`);
  });
  
  ws.on('error', (error) => {
    console.error('WebSocket error:', error.message);
  });
});

const PORT = process.env.PORT || 9090;
server.listen(PORT, '0.0.0.0', () => {
  console.log(`YYC3 Heartbeat Relay running on port ${PORT}`);
  console.log(`WebSocket endpoint: ws://0.0.0.0:${PORT}/ws/heartbeat`);
});
